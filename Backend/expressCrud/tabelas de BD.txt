CREATE TABLE usuarios (idUsuario INTEGER PRIMARY KEY AUTO_INCREMENT, nomeCliente VARCHAR(255) not null, cpf VARCHAR(11) UNIQUE not null, logradouro VARCHAR(100) NOT NULL, numero int not null, bairro VARCHAR(30) NOT NULL, cidade VARCHAR(50) NOT NULL,  emailCliente VARCHAR(100) not null, senha VARCHAR(10) not null, tipo TINYINT not null) AUTO_INCREMENT= 201;
